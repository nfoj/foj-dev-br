Mostra Diretório Atual


 // Entrada
 pwd

 /home/user

Tab: autocompleta


 tab + "name"

Lista arquivos e diretórios


 ls 

 ls *

 ls -a

 la -al

 // Saída com tamanho dos arquivos
 ls -lh "name"

Limpa a tela: clear or ctrl + l


 clear
 ctrl + l

 ls --help

FHS: Hierarquia de Sistemas de Arquivos (Filesystem Hierarchy Standard): cd 


 // Entrada
 cd

 // Saída
 home/user
        
 // Entrada
 cd .config

 // Saída
 alacritty  dconf  gtk-3.0
        
 // Entrada
 cd /etc/alsa/
 
 // Saída
 conf.d
        
 // retorna
 cd -
 
 // Saída
 cd /etc
        
 // próximo 
 cd ../.config/

Cria diretórios: mkdir


 // Cria diretórios
 mdkir "nome"

 // Cria mais de um diretório
 mdkir "nome"

 // nome + espaço
 mkdir "name"\ "nome"
              
 // Cria diretórios um dentro do outro
 mkdir -p dir/dir/dir1/dir2  
            
 // Cria diretórios ocultos
 mdkir ."nome"

Cria arquivos: touch ou >


touch "nome"
touch Texto.txt
                
> "nome"
> Texto.txt
            
// Cria multiplos arquivos
touch "nome_01" "nome_02" "nome_03"
            
// Cria arquivos ocultos
touch ."nome"

Remove: rm "nome" ou rm -r "nome" ou rm -rf "nome"


 // remove arquivos
 rm "nome"

 // remove diretórios 
 rm -r 

 // deleta todos arquivos e links
 rm -rf

Cópia arquivos ou diretórios: cp


 cp "nome" path

 // cp todos
 cp -r * path
 cp -r * ../dir2/dir3

Globbing (Manipulação de arquivos): *, ? and [ ]


 // *: substitui caracteres
 ls arq* 
 ls *1*
          
 // ? = substitui caracteres
 ls arq1?
 ls arq??
        
 // [] = seleciona um intervalo
 ls arq[1-5]
 ls *[1-5]
 ls arq[1,2,5]
 ls ???[1-5]
 ls ???[1-5]*
 ls [A,a]rq[1-5]
