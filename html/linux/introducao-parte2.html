Linux - Introdução - Parte 2

Ler arquivos: cat


 cat "nome"

Procura dados em arquivos: grep


 grep "name/character/" "local"
 grep http service
                        
 // Ignora case-sensitive: -i
 grep -i http service
                        
 // Retorna o nome dos arquivos 
 // Critérios especificados
 grep -l http *  
                        
 // Retorna o nome dos arquivos 
 // Critérios não especificados
 grep -L http *
                       
 // Recursivo: -r
 grep -r http *
                        
 // Recursivo + Lista
 grep -rl http *
                        
 // Recursivo + case-insensitive
 grep -ri http *

Fornece uma maneira mais eficiente de visualizar arquivos grandes do que cat: less


 // Setas do Teclado 
 less "nome"
                  
 // Espaço: scroll na tela
 // Enter: pula linha a linha
 // b: scrolla para cima
 // q: sai
 more "nome"
      
 // Mostra das primeiras linhas
 head "nome"
                 
 // Mostras as 5 primeiras linhas
 head -n 5 "nome"
                 
 // Mostra as últimas linhas
 tail "nome"
                 
 // Mostras as 5 últimas linhas
 tail -n 5 "nome"

Pesquisa arquivos: find


 find / -nome *.conf

 // Super usuário
 sudo find / -nome *.conf
                 
 // maxdeath (conta diretórios) "nome"
 find /etc -maxdepth 2 -nome *.conf    
                 
 // Tempo - Minutos
 find . -amin -5
                 
 // Tempo - Dias
 find . -atime -2
                 
 // Tamanho (B, M, G)
 find / -size +100M
 find / -size +3G

Copia path/files/ search: > , >>


 // Copia e substitui: >
 grep ssh service > text.txt
 cat /etc/passwd > text.txt
                     
 // Copia a adiciona >>
 grep ssh services >> text.txt 
                 
 // Mais de um comendando: |
 cat /etc/passwd | grep "nome" > list.txt

Sort: sort


 cat /etc/passwd
 cat text.txt | sort
 cat text.txt | sort > list.txt

Filtro: cut -d 


 //cut -d " " -f"número da coluna" "nome"
 cut -d " " -f1 list.txt
                        
 cat list.txt | cut -d " " -f1 
 cat list.txt | cut -d " " -f1- 
 cat list.txt | cut -d " " -f1-2 > text1.txt
 cat list.txt | cut -d " " -f1,4- > text2.txt

Expresão: -E


 // ^ início do arquivo
 cat "nome" | grep -E "^A"

 // $ final do arquivo
 cat "nome" | grep -E "a$"

 // Inicio e final
 cat "nome" | grep -E "^a$"

 // case-insensitive
 cat "nome" | grep -iE "a"

 // . : todos arquivos
 cat "nome" | grep -iE "^.at"

 // [abc] [124]
 cat "nome" | grep -iE "^[20]..
