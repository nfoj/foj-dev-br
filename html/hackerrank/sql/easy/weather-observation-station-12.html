Problem:

Query the list of CITY names from STATION that do not end with vowels. Your result cannot contain duplicates.

Field Type
ID NUMBER
CITY VARCHAR2(21)
STATE VARCHAR2(2)
LAT_N NUMBER
LAT_W NUMBER

where LAT_N is the northern latitude and LONG_W is the western longitude.

# Solution

            
    SELECT DISTINCT CITY
    FROM STATION
    WHERE RIGHT(LOWER(CITY), 1) NOT IN ('a', 'e', 'i', 'o', 'u')
     AND  LEFT(LOWER(CITY), 1) NOT IN ('a', 'e', 'i', 'o', 'u')
    ORDER BY CITY ASC;

# Solution_01

            
    SELECT DISTINCT CITY
    FROM STATION
    WHERE LOWER(SUBSTR(CITY, 1, 1)) NOT IN ('a', 'e', 'i', 'o', 'u')
       OR LOWER(SUBSTR(CITY, LENGTH(CITY), 1)) NOT IN ('a', 'e', 'i', 'o', 'u')
    ORDER BY CITY ASC;

# Solution_02

            
    SELECT DISTINCT CITY
    FROM STATION
    WHERE SUBSTRING(City,1,1) NOT IN ('a','e','o','i','u')
       OR SUBSTRING(City,-1) NOT IN ('a','e','o','i','u')
    ORDER BY CITY ASC;
